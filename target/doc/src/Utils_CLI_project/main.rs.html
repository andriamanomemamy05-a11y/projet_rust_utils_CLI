12/// Point d'entrée de l'application.
13///
14/// Cette fonction affiche un menu interactif permettant de choisir un utilitaire Linux
15/// parmi : ls, cat, cp, mv, rm, wc. L'utilisateur peut entrer le numéro correspondant
16/// ou `quit` pour quitter l'application.
17///
18/// L'application boucle tant que l'utilisateur ne choisit pas de quitter.
19/// Elle gère les entrées invalides en affichant un message d'erreur.
20///
21/// # Algorithme
22///  - Boucle infinie jusqu'à ce que l'utilisateur tape `quit`.
23///  - Affiche le menu avec les différentes options disponibles.
24///  - Demande à l'utilisateur de saisir son choix.
25///  - Lit l'entrée utilisateur et supprime les espaces et retours à la ligne.
26///  - Vérifie si l'entrée est `quit` : si oui, quitte la boucle et termine le programme.
27///  - Sinon, effectue un `match` sur la saisie :
28///    - `"1"` : appelle la fonction `ls::ls()`
29///    - `"2"` : appelle la fonction `cat::cat()`
30///    - `"3"` : appelle la fonction `cp::cp()`
31///    - `"4"` : appelle la fonction `mv::mv()`
32///    - `"5"` : appelle la fonction `rm::rm()`
33///    - `"6"` : appelle la fonction `wc::wc()`
34///    - Autre : affiche un message d'erreur et redemande une saisie.
35///  - Affiche une ligne vide pour séparer les itérations.
36///
37/// Ce processus se répète jusqu'à ce que l'utilisateur décide de quitter.
38pub fn main() {
39    loop {
40        // Affichage du menu
41        println!("Bonjour et bienvenue dans l'utilitaire de commande linux.");
42        println!();
43        println!("Veuillez choisir votre utilitaire (tapez quit pour quitter) :");
44        println!("1) ls");
45        println!("2) cat");
46        println!("3) cp");
47        println!("4) mv");
48        println!("5) rm");
49        println!("6) wc");
50        println!();
51        print!("Votre choix : ");
52        io::stdout().flush().unwrap(); // Assure que le prompt s'affiche avant la saisie
53
54        // Lecture de l'entrée utilisateur
55        let mut choix = String::new();
56        io::stdin()
57            .read_line(&mut choix)
58            .expect("Erreur lors de la lecture de l'entrée");
59        let choix = choix.trim(); // Supprime les espaces et le retour à la ligne
60
61        // Gestion de la commande "quit"
62        if choix.eq_ignore_ascii_case("quit") {
63            println!("A bientôt !");
64            break;
65        }
66
67        // Match sur l'entrée utilisateur
68        match choix {
69            "1" => {
70                println!("Exécution de ls...");
71                //ls::ls(); // Appel de la fonction ls (à implémenter)
72            }
73            "2" => {
74                println!("Exécution de cat...");
75                cat::cat(); // Appel de la fonction cat
76            }
77            "3" => {
78                println!("Exécution de cp...");
79                //cp::cp(); // Appel de la fonction cp (à implémenter)
80            }
81            "4" => {
82                println!("Exécution de mv...");
83                //mv::mv(); // Appel de la fonction mv (à implémenter)
84            }
85            "5" => {
86                println!("Exécution de rm...");
87                //rm::rm(); // Appel de la fonction rm (à implémenter)
88            }
89            "6" => {
90                println!("Exécution de wc...");
91                wc::wc(); // Appel de la fonction wc (à implémenter)
92            }
93            _ => {
94                // Gestion des entrées invalides
95                println!("Option invalide, veuillez réessayer !");
96            }
97        }
98
99        println!(); // Ligne vide avant le prochain tour
100    }
101}
