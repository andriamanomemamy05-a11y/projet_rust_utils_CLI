27    /// Fichier contenant une liste de fichiers séparés par `\0` (`--files0-from=F`).

48/// Fonction principale du programme `wc`.
49///
50/// Démarre un shell interactif permettant de saisir des commandes `wc`. Si l'utilisateur
51/// quitte l'utilitaire en tapant `quit`, il reviendra sur le menu principal.
52///
53/// # Exemple
54/// ```no_run
55/// wc();
56/// ```
57pub fn wc() {
58    loop {
59        println!("\n=== Programme utilitaire wc ===");
60        println!("Entrez votre commande (ou 'quit' pour quitter) :");
61        print!("> ");
62        io::stdout().flush().unwrap();
63
64        let mut input = String::new();
65        io::stdin().read_line(&mut input).expect("Erreur lors de la lecture de l'entrée");
66        
67        let input = input.trim();
68        
69        if input == "quit" {
70            break;
71        }
72
73        if input.is_empty() {
74            continue;
75        }
76
77        // Vérifier si la commande contient un pipe
78        if input.contains('|') {
79            // Traiter la commande avec pipe (echo ... | wc ...)
80            match process_piped_command(input) {
81                Ok(_) => {},
82                Err(e) => {
83                    eprintln!("Erreur : {}", e);
84                    eprintln!("Stack trace : {:?}", e);
85                }
86            }
87        } else {
88            // Traiter la commande simple (wc ...)
89            let parts = parse_command_line(input);
90            
91            if parts.is_empty() {
92                continue;
93            }
94
95            // Vérifier que la commande commence par wc
96            if parts[0] != "wc" {
97                println!("Erreur : La commande doit commencer par 'wc'");
98                continue;
99            }
100
101            // Traiter la commande
102            let args: Vec<&str> = parts[1..].iter().map(|s| s.as_str()).collect();
103            match process_command(&args) {
104                Ok(_) => {},
105                Err(e) => {
106                    eprintln!("Erreur : {}", e);
107                    eprintln!("Stack trace : {:?}", e);
108                }
109            }
110        }
111    }
112}

114/// Parse une ligne de commande en respectant les guillemets et échappements.
115///
116/// # Algorithme
117/// - Parcours caractère par caractère.
118/// - Bascule l'état `in_quotes` sur les guillemets.
119/// - Si espace ou tabulation et hors guillemets, finalise le mot.
120/// - Gère les échappements `\`.
121/// - Ajoute le mot final à la liste.
122///
123/// # Arguments
124/// * `input` - Ligne de commande brute.
125///
126/// # Retour
127/// Vecteur de chaînes (`Vec<String>`), chaque élément un argument.
128///
129/// # Exemple
130/// ```rust
131/// let args = wc_rs::parse_command_line(r#"wc -l "fichier avec espaces.txt""#);
132/// assert_eq!(args, vec!["wc", "-l", "fichier avec espaces.txt"]);
133/// ```
134fn parse_command_line(input: &str) -> Vec<String> {
135    let mut result = Vec::new();
136    let mut current = String::new();
137    let mut in_quotes = false;
138    let mut chars = input.chars().peekable();
139
140    while let Some(ch) = chars.next() {
141        match ch {
142            '"' => {
143                in_quotes = !in_quotes;
144            },
145            ' ' | '\t' => {
146                if in_quotes {
147                    current.push(ch);
148                } else if !current.is_empty() {
149                    result.push(current.clone());
150                    current.clear();
151                }
152            },
153            '\\' => {
154                // Gérer les échappements
155                if let Some(&next_ch) = chars.peek() {
156                    if next_ch == '"' || next_ch == '\\' {
157                        current.push(chars.next().unwrap());
158                    } else {
159                        current.push(ch);
160                    }
161                } else {
162                    current.push(ch);
163                }
164            },
165            _ => {
166                current.push(ch);
167            }
168        }
169    }
170
171    if !current.is_empty() {
172        result.push(current);
173    }
174
175    result
176}

178/// Interprète les séquences d'échappement dans une chaîne.
179///
180/// # Algorithme
181/// - Parcours de la chaîne caractère par caractère.
182/// - Si `\`, détermine la séquence (`n`, `t`, `r`, `xFF`, etc...).
183/// - Remplace par le caractère correspondant.
184///
185/// # Arguments
186/// * `input` - Chaîne avec séquences d'échappement.
187///
188/// # Retour
189/// Chaîne transformée.
190///
191/// # Exemple
192/// ```rust
193/// let text = wc_rs::unescape("Hello\\nWorld");
194/// assert_eq!(text, "Hello\nWorld");
195/// ```
196fn unescape(input: &str) -> String {
197    let mut output = String::new();
198    let mut chars = input.chars().peekable();
199
200    while let Some(c) = chars.next() {
201        if c == '\\' {
202            match chars.peek() {
203                Some('n') => { output.push('\n'); chars.next(); }
204                Some('t') => { output.push('\t'); chars.next(); }
205                Some('r') => { output.push('\r'); chars.next(); }
206                Some('v') => { output.push('\u{000B}'); chars.next(); }
207                Some('a') => { output.push('\u{0007}'); chars.next(); }
208                Some('x') => {
209                    chars.next();
210                    let hex: String = chars.by_ref().take(2).collect();
211                    if let Ok(val) = u8::from_str_radix(&hex, 16) {
212                        output.push(val as char);
213                    }
214                }
215                Some('\\') => { output.push('\\'); chars.next(); }
216                _ => output.push(c),
217            }
218        } else {
219            output.push(c);
220        }
221    }
222
223    output
224}

226/// Traite une commande contenant un pipe (`echo ... | wc ...`).
227///
228/// # Algorithme
229/// 1. Sépare par `|`.
230/// 2. Vérifie la partie echo.
231/// 3. Extrait le texte après echo.
232/// 4. Vérifie la partie wc.
233/// 5. Parse les options.
234/// 6. Applique le comptage sur le texte.
235///
236/// # Arguments
237/// * `input` - Ligne de commande avec pipe.
238///
239/// # Retour
240/// `io::Result<()>` indiquant succès ou erreur.
241///
242/// # Exemple
243/// ```no_run
244/// process_piped_command(r#"echo "Hello World" | wc -w"#)?;
245/// // Affiche : 2
246/// ```
247fn process_piped_command(input: &str) -> io::Result<()> {
248    let pipe_parts: Vec<&str> = input.split('|').map(|s| s.trim()).collect();
249    
250    if pipe_parts.len() != 2 {
251        return Err(io::Error::new(
252            io::ErrorKind::InvalidInput,
253            "Format invalide : utilisez 'echo [texte] | wc [options]'"
254        ));
255    }
256
257    let echo_part = pipe_parts[0];
258    let wc_part = pipe_parts[1];
259
260    // Parser la partie echo avec gestion des guillemets
261    let echo_parsed = parse_command_line(echo_part);
262    if echo_parsed.is_empty() || echo_parsed[0] != "echo" {
263        return Err(io::Error::new(
264            io::ErrorKind::InvalidInput,
265            "La commande doit commencer par 'echo'"
266        ));
267    }
268
269    // Extraire le texte après echo (tout sauf le premier mot "echo")
270    let stdin_text = echo_parsed[1..].join(" ");
271
272    // Parser la partie wc avec gestion des guillemets
273    let wc_parsed = parse_command_line(wc_part);
274    if wc_parsed.is_empty() || wc_parsed[0] != "wc" {
275        return Err(io::Error::new(
276            io::ErrorKind::InvalidInput,
277            "Après le pipe, la commande doit être 'wc [options]'"
278        ));
279    }
280
281    // Parser les options de wc (tout sauf le premier mot "wc")
282    let wc_args: Vec<&str> = wc_parsed[1..].iter().map(|s| s.as_str()).collect();
283    let (options, _) = parse_arguments(&wc_args)?;
284
285    // Traiter le stdin avec les options
286    let text_unescape = unescape(&stdin_text);
287    process_stdin(&text_unescape, &options)?;
288
289    Ok(())
290}

292/// Traite une commande `wc` avec ses arguments.
293///
294/// # Algorithme
295/// - Gère `--help` et `--version`.
296/// - Parse les options et les fichiers.
297/// - Traite le(s) fichier(s) ou stdin.
298///
299/// # Arguments
300/// * `args` - Arguments de la commande.
301///
302/// # Retour
303/// `io::Result<()>` indiquant succès ou erreur.
304///
305/// # Exemple
306/// ```no_run
307/// process_command(&["-l", "fichier.txt"])?;
308/// // Affiche : 42 fichier.txt
309/// ```
310fn process_command(args: &[&str]) -> io::Result<()> {
311    // Gérer --help
312    if args.contains(&"--help") {
313        display_help();
314        return Ok(());
315    }
316
317    // Gérer --version
318    if args.contains(&"--version") {
319        display_version();
320        return Ok(());
321    }
322
323    // Parser les options et les fichiers
324    let (options, file_paths) = parse_arguments(args)?;
325
326    // Si aucun fichier spécifié, erreur
327    if file_paths.is_empty() {
328        println!("Erreur : Aucun fichier spécifié");
329        println!("Utilisez 'wc --help' pour plus d'informations");
330        return Ok(());
331    }
332
333    // Traiter les fichiers
334    let mut total = CountResult::default();
335    let multiple_files = file_paths.len() > 1;
336
337    for path in &file_paths {
338        match process_file(path) {
339            Ok(result) => {
340                display_result(&result, &options, Some(path));
341                
342                // Accumuler pour le total
343                if multiple_files {
344                    total.lines += result.lines;
345                    total.words += result.words;
346                    total.chars += result.chars;
347                    total.bytes += result.bytes;
348                    total.max_line_length = total.max_line_length.max(result.max_line_length);
349                }
350            },
351            Err(e) => {
352                eprintln!("wc: {}: {}", path, e);
353            }
354        }
355    }
356
357    // Afficher le total si plusieurs fichiers
358    if multiple_files {
359        display_result(&total, &options, Some("total"));
360    }
361
362    Ok(())
363}

365/// Parse les arguments pour extraire options et chemins de fichiers.
366///
367/// # Algorithme
368/// - Pour chaque argument :
369///   - S'il commence par `-`, est traité comme option.
370///   - Sinon, est considéré comme un chemin fichier.
371///
372/// # Arguments
373/// * `args` - Slice des arguments.
374///
375/// # Retour
376/// Tuple `(Options, Vec<String>)`.
377///
378/// # Exemple
379/// ```rust
380/// let (opts, files) = wc_rs::parse_arguments(&["-l", "fichier.txt"]).unwrap();
381/// assert_eq!(files[0], "fichier.txt");
382/// ```
383fn parse_arguments(args: &[&str]) -> io::Result<(Options, Vec<String>)> {
384    let mut options = Options::default();
385    let mut file_paths: Vec<String> = Vec::new();
386
387    let mut i = 0;
388    while i < args.len() {
389        let arg = args[i];
390        
391        if arg.starts_with("--files0-from=") {
392            let file = arg.trim_start_matches("--files0-from=");
393            options.files0_from = Some(file.to_string());
394        } else if arg.starts_with('-') {
395            parse_option(arg, &mut options)?;
396        } else {
397            file_paths.push(arg.to_string());
398        }
399        
400        i += 1;
401    }
402
403    // Si aucune option de comptage n'est spécifiée, afficher lignes, mots et octets (comportement par défaut)
404    if !options.show_bytes && !options.show_chars && !options.show_lines 
405        && !options.show_words && !options.show_max_line_length {
406        options.show_lines = true;
407        options.show_words = true;
408        options.show_bytes = true;
409    }
410
411    Ok((options, file_paths))
412}

414/// Parse une option et met à jour la structure `Options`.
415///
416/// # Algorithme
417/// - Vérifie si l'option est simple ou combinée.
418/// - Met à jour les champs correspondants dans `Options`.
419///
420/// # Arguments
421/// * `opt` - Chaîne représentant l'option.
422/// * `options` - Référence mutable de `Options`.
423///
424/// # Retour
425/// `io::Result<()>` indiquant succès ou erreur.
426///
427/// # Exemple
428/// ```rust
429/// let mut opts = wc_rs::Options::default();
430/// wc_rs::parse_option("-l", &mut opts).unwrap();
431/// assert!(opts.show_lines);
432/// ```
433fn parse_option(opt: &str, options: &mut Options) -> io::Result<()> {
434    match opt {
435        "-c" | "--bytes" => options.show_bytes = true,
436        "-m" | "--chars" => options.show_chars = true,
437        "-l" | "--lines" => options.show_lines = true,
438        "-w" | "--words" => options.show_words = true,
439        "-L" | "--max-line-length" => options.show_max_line_length = true,
440        _ => {
441            // Gérer les options combinées (ex: -lwc)
442            if opt.starts_with('-') && opt.len() > 2 && !opt.starts_with("--") {
443                for c in opt[1..].chars() {
444                    let single_opt = format!("-{}", c);
445                    parse_option(&single_opt, options)?;
446                }
447            } else if !opt.starts_with("--files0-from=") {
448                return Err(io::Error::new(
449                    io::ErrorKind::InvalidInput,
450                    format!("Option invalide : {}", opt)
451                ));
452            }
453        }
454    }
455    Ok(())
456}

458/// Traite un fichier ligne par ligne.
459///
460/// # Algorithme
461/// - Vérifie l'existence du fichier.
462/// - Lit le fichier ligne par ligne avec BufRead.
463/// - Compte les lignes, mots, caractères et octets au fur et à mesure.
464/// - Trouve la ligne la plus longue.
465/// - Retourne les résultats.
466///
467/// # Arguments
468/// * `file_path` - Chemin vers le fichier.
469///
470/// # Retour
471/// `io::Result<CountResult>`.
472///
473/// # Exemple
474/// ```no_run
475/// let result = process_file("fichier.txt")?;
476/// println!("Lignes: {}", result.lines);
477/// ```
478fn process_file(file_path: &str) -> io::Result<CountResult> {
479    // Vérifier si le fichier existe
480    if !Path::new(file_path).exists() {
481        return Err(io::Error::new(
482            io::ErrorKind::NotFound,
483            format!("Aucun fichier ou dossier de ce type")
484        ));
485    }
486
487    // Ouvrir le fichier
488    let file = File::open(file_path)?;
489    let reader = BufReader::new(file);
490    
491    let mut result = CountResult::default();
492    
493    // Traiter ligne par ligne
494    for line_result in reader.lines() {
495        let line = line_result?;
496        
497        // Compter les lignes
498        result.lines += 1;
499        
500        // Compter les mots dans la ligne
501        result.words += line.split_whitespace().count();
502        
503        // Compter les caractères dans la ligne (+ 1 pour le \n)
504        result.chars += line.chars().count() + 1;
505        
506        // Compter les octets dans la ligne (+ 1 pour le \n)
507        result.bytes += line.as_bytes().len() + 1;
508        
509        // Trouver la longueur maximale
510        let line_length = line.chars().count();
511        if line_length > result.max_line_length {
512            result.max_line_length = line_length;
513        }
514    }
515
516    Ok(result)
517}

537/// Compte les lignes, mots, caractères et longueur maximale dans un contenu.
538///
539/// # Algorithme
540/// 1. Compte les lignes en parcourant le texte.
541/// 2. Compte les mots en séparant par les espaces blancs.
542/// 3. Compte les caractères Unicode.
543/// 4. Trouve la ligne la plus longue.
544///
545/// # Arguments
546/// * `content` - Texte à analyser.
547///
548/// # Retour
549/// `CountResult` avec toutes les statistiques.
550///
551/// # Exemple
552/// ```rust
553/// use wc_rs::count_content;
554/// let result = count_content("Hello World\nBonjour");
555/// assert_eq!(result.lines, 2);
556/// assert_eq!(result.words, 3);
557/// ```
558fn count_content(content: &str) -> CountResult {
559    let mut result = CountResult::default();
560
561    // Compter les lignes et trouver la ligne la plus longue
562    let lines: Vec<&str> = content.lines().collect();
563    result.lines = lines.len();
564    
565    // Si le contenu se termine par un retour à la ligne, on compte cette ligne vide
566    if content.ends_with('\n') && !content.is_empty() {
567        result.lines += 1;
568    }
569
570    for line in &lines {
571        let line_length = line.chars().count();
572        if line_length > result.max_line_length {
573            result.max_line_length = line_length;
574        }
575    }
576
577    // Compter les mots (séparés par des espaces blancs)
578    result.words = content.split_whitespace().count();
579
580    // Compter les caractères Unicode
581    result.chars = content.chars().count();
582
583    // Compter les octets
584    result.bytes = content.as_bytes().len();
585
586    result
587}

590/// Affiche les résultats du comptage selon les options.
591///
592/// # Algorithme
593/// - Affiche chaque compteur demandé dans l'ordre : lignes, mots, caractères/octets, longueur max.
594/// - Ajoute le nom du fichier si fourni.
595///
596/// # Arguments
597/// * `result` - Résultats du comptage.
598/// * `options` - Options indiquant quoi afficher.
599/// * `filename` - Nom du fichier optionnel.
600///
601/// # Exemple
602/// ```rust
603/// let result = CountResult { lines: 10, words: 50, chars: 200, bytes: 200, max_line_length: 80 };
604/// display_result(&result, &options, Some("fichier.txt"));
605/// // Affiche : 10 50 200 fichier.txt
606/// ```
607fn display_result(result: &CountResult, options: &Options, filename: Option<&str>) {
608    let mut output = String::new();
609
610    if options.show_lines {
611        output.push_str(&format!("{:7} ", result.lines));
612    }
613
614    if options.show_words {
615        output.push_str(&format!("{:7} ", result.words));
616    }
617
618    // Si -m et -c sont tous les deux spécifiés, -m prend la priorité
619    if options.show_chars {
620        output.push_str(&format!("{:7} ", result.chars));
621    } else if options.show_bytes {
622        output.push_str(&format!("{:7} ", result.bytes));
623    }
624
625    if options.show_max_line_length {
626        output.push_str(&format!("{:7} ", result.max_line_length));
627    }
628
629    if let Some(name) = filename {
630        output.push_str(name);
631    }
632
633    println!("{}", output.trim_end());
634}
